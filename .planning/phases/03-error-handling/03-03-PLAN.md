    **A. Add API key highlight/deep-link support to settings.html.**

    The skill execution error from Plan 02 links to `/settings?tab=api-keys&highlight=openrouter`. The settings page needs to:
    1. Switch to the API Keys tab (already works via `?tab=api-keys` — the inline JS reads it)
    2. Scroll to the specific API key item and highlight it with a visual pulse

    Add this to the inline `<script>` block in settings.html (at the end, before the closing `</script>`):

    ```javascript
    // Deep-link: highlight a specific API key if ?highlight=keyname is present
    (function() {
        var params = new URLSearchParams(window.location.search);
        var highlight = params.get('highlight');
        var tab = params.get('tab');

        // Switch to API keys tab if specified
        if (tab) {
            document.querySelectorAll('.settings-tab').forEach(function(t) { t.classList.remove('active'); });
            document.querySelectorAll('.settings-panel').forEach(function(p) { p.classList.remove('active'); });
            var targetTab = document.querySelector('.settings-tab[data-tab="' + tab + '"]');
            var targetPanel = document.getElementById('panel-' + tab);
            if (targetTab) targetTab.classList.add('active');
            if (targetPanel) targetPanel.classList.add('active');
        }

        // Highlight specific API key
        if (highlight) {
            var keyItem = document.querySelector('.api-key-item[data-key="' + highlight + '"]');
            if (keyItem) {
                // Scroll into view
                setTimeout(function() {
                    keyItem.scrollIntoView({ behavior: 'smooth', block: 'center' });
                }, 200);

                // Add highlight animation
                keyItem.classList.add('api-key-highlighted');

                // Focus the input field
                var input = keyItem.querySelector('input');
                if (input) {
                    setTimeout(function() { input.focus(); }, 500);
                }

                // Remove highlight after animation
                setTimeout(function() {
                    keyItem.classList.remove('api-key-highlighted');
                }, 4000);
            }
        }
    })();
    ```

    Add CSS for the highlight animation in the `<style>` block of settings.html:

    ```css
    /* API key highlight animation (for deep-linking from error pages) */
    .api-key-highlighted {
        animation: keyHighlightPulse 2s ease-in-out 2;
        box-shadow: 0 0 0 2px var(--accent);
    }

    @keyframes keyHighlightPulse {
        0%, 100% { box-shadow: 0 0 0 2px var(--accent); }
        50% { box-shadow: 0 0 0 4px var(--accent), 0 0 12px rgba(var(--accent-rgb, 99, 102, 241), 0.3); }
    }
    ```
