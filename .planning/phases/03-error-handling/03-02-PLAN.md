    ```javascript
    // Error classification and recovery guidance
    function classifyError(status, data) {
        var message = data.message || data.error || 'An error occurred';
        var messageLower = message.toLowerCase();
        var recovery = [];

        // Missing API key detection
        if (messageLower.includes('api key') || messageLower.includes('api_key') ||
            messageLower.includes('not configured') || messageLower.includes('authentication') ||
            messageLower.includes('unauthorized') || status === 401) {

            // Try to detect which key is missing
            var keyHints = {
                'openrouter': 'openrouter',
                'perplexity': 'perplexity',
                'anthropic': 'anthropic',
                'openai': 'openai',
                'slack': 'slack',
                'fal': 'fal'
            };

            var detectedKey = '';
            for (var hint in keyHints) {
                if (messageLower.includes(hint)) {
                    detectedKey = keyHints[hint];
                    break;
                }
            }

            var settingsUrl = '/settings?tab=api-keys' + (detectedKey ? '&highlight=' + detectedKey : '');

            return {
                title: 'API Key Required',
                message: message,
                icon: 'key',
                recoveryHTML: '<a href="' + settingsUrl + '" class="error-action-link">Go to Settings to configure your API key</a>'
            };
        }

        // Validation errors (field-level)
        if (status === 400 && data.errors) {
            var fieldList = Object.entries(data.errors).map(function(entry) {
                return '<li><strong>' + escapeHtml(entry[0]) + ':</strong> ' + escapeHtml(entry[1]) + '</li>';
            }).join('');

            return {
                title: 'Validation Error',
                message: 'Please fix the following fields:',
                icon: 'alert',
                recoveryHTML: '<ul class="error-field-list">' + fieldList + '</ul>'
            };
        }

        // Rate limiting
        if (status === 429) {
            return {
                title: 'Too Many Requests',
                message: 'The service is temporarily overloaded. Please wait a moment.',
                icon: 'clock',
                recoveryHTML: '<p>Wait 30 seconds and try again. If this persists, check your API provider dashboard for rate limits.</p>'
            };
        }

        // Server errors
        if (status >= 500) {
            return {
                title: 'Server Error',
                message: message,
                icon: 'server',
                recoveryHTML: '<p>This is usually temporary. Try again in a moment.</p><p>If the problem persists, check that your API keys are valid in <a href="/settings?tab=api-keys">Settings</a>.</p>'
            };
        }

        // Skill not found
        if (status === 404) {
            return {
                title: 'Skill Not Found',
                message: message,
                icon: 'search',
                recoveryHTML: '<a href="/skills" class="error-action-link">Browse available skills</a>'
            };
        }

        // Network / timeout (status 0)
        if (status === 0) {
            return {
                title: 'Connection Problem',
                message: message,
                icon: 'wifi',
                recoveryHTML: '<p>Check your internet connection and try again.</p>'
            };
        }

        // Generic fallback
        return {
            title: 'Execution Failed',
            message: message,
            icon: 'alert',
            recoveryHTML: '<p>Try again. If the problem persists, <a href="/help">visit the Help page</a>.</p>'
        };
    }

    function showExecutionError(status, data) {
        var classified = classifyError(status, data);

        // Also highlight field-level errors in the form
        if (data.errors) {
            Object.keys(data.errors).forEach(function(fieldName) {
                var group = document.querySelector('[data-field="' + fieldName + '"]');
                if (group) {
                    showFieldValidationError(group, data.errors[fieldName]);
                }
            });
        }

        // Show error panel below the form
        var panel = document.getElementById('execution-error-panel');
        if (!panel) {
            panel = document.createElement('div');
            panel.id = 'execution-error-panel';
            panel.className = 'execution-error-panel';
            // Insert after the form card
            var formCard = document.getElementById('form-mode-container');
            formCard.parentNode.insertBefore(panel, formCard.nextSibling);
        }

        panel.innerHTML = '<div class="execution-error-header">' +
            '<strong>' + escapeHtml(classified.title) + '</strong>' +
            '<button class="toast-close" onclick="hideExecutionError()" style="background:none;border:none;cursor:pointer;color:inherit;">' +
            '<svg width="16" height="16" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><line x1="18" y1="6" x2="6" y2="18"/><line x1="6" y1="6" x2="18" y2="18"/></svg>' +
            '</button></div>' +
            '<p class="execution-error-message">' + escapeHtml(classified.message) + '</p>' +
            '<div class="execution-error-recovery">' + classified.recoveryHTML + '</div>';

        panel.style.display = 'block';
        panel.scrollIntoView({ behavior: 'smooth', block: 'nearest' });
    }

    window.hideExecutionError = function() {
        var panel = document.getElementById('execution-error-panel');
        if (panel) panel.style.display = 'none';
    };
    ```
