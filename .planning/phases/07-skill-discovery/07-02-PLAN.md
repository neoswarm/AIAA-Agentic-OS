3. **Replace the search input handler** in the JS IIFE. Find the `searchInput.addEventListener('input', ...)` handler (around line 810) and replace it so that instead of filtering allSkills locally with `.includes()`, it calls the backend API `/api/v2/skills/search?q=` with 300ms debounce. The new handler:
```javascript
searchInput.addEventListener('input', function() {
    clearTimeout(searchTimeout);
    highlightIndex = -1;
    var q = this.value.trim();
    if (q.length < 2) {
        searchResults.classList.remove('visible');
        return;
    }
    searchTimeout = setTimeout(async function() {
        try {
            var res = await fetch('/api/v2/skills/search?q=' + encodeURIComponent(q));
            if (!res.ok) return;
            var data = await res.json();
            var matches = (data.skills || []).slice(0, 8);

            if (matches.length === 0) {
                searchResults.innerHTML = '<div style="padding: 1.25rem; text-align: center;"><div style="color: var(--text-muted); font-size: 0.875rem; margin-bottom: 0.25rem;">No skills match your search</div><div style="color: var(--text-muted); font-size: 0.75rem;">Try different keywords or <a href="/skills" style="color: var(--accent);">browse all skills</a></div></div>';
            } else {
                searchResults.innerHTML = '';
                matches.forEach(function(s, i) {
                    var item = document.createElement('a');
                    item.className = 'search-result-item';
                    item.href = '/skills/' + encodeURIComponent(s.name) + '/run';
                    item.setAttribute('data-index', i);

                    var metaHtml = '<div class="result-meta">';
                    if (s.estimated_minutes) {
                        metaHtml += '<span class="time-estimate">' + s.estimated_minutes + ' min</span>';
                    }
                    if (s.complexity) {
                        metaHtml += '<span class="complexity-badge complexity-' + escapeHtml(s.complexity) + '">' + escapeHtml(s.complexity) + '</span>';
                    }
                    metaHtml += '</div>';

                    item.innerHTML = '<div style="flex:1; min-width:0;">'
                        + '<div class="result-name">' + escapeHtml(formatSkillName(s.name)) + '</div>'
                        + '<div class="result-desc">' + escapeHtml(s.description || '') + '</div>'
                        + metaHtml
                        + '</div>';
                    searchResults.appendChild(item);
                });
            }
            searchResults.classList.add('visible');
        } catch (e) {
            console.warn('Search error:', e);
        }
    }, 300);
});
```

IMPORTANT: This replaces the old client-side `.includes()` filter with the API call. The `allSkills` array is still loaded for category rendering -- do NOT remove the loadSkills() function. The search just now uses the enhanced backend instead of local filtering.
IMPORTANT: All text inserted via innerHTML MUST use `escapeHtml()` for any data from the API response.

async function loadRecommended() {
    try {
        var res = await fetch('/api/v2/skills/recommended');
        if (!res.ok) return;
        var data = await res.json();
        var skills = (data.skills || []).slice(0, 8);
        if (skills.length === 0) return;

        var grid = document.getElementById('recommended-grid');
        grid.innerHTML = '';
        skills.forEach(function(s) {
            var card = document.createElement('a');
            card.className = 'quick-start-card';
            card.href = '/skills/' + encodeURIComponent(s.name) + '/run';
            card.style.position = 'relative';

            var metaLine = '';
            if (s.estimated_minutes) {
                metaLine += '<span class="time-estimate">' + s.estimated_minutes + ' min</span> ';
            }
            if (s.complexity) {
                metaLine += '<span class="complexity-badge complexity-' + escapeHtml(s.complexity) + '">' + escapeHtml(s.complexity) + '</span>';
            }

            card.innerHTML = '<div class="quick-start-icon" style="background: var(--accent);">'
                + '<svg viewBox="0 0 24 24" stroke-width="2"><polygon points="13 2 3 14 12 14 11 22 21 10 12 10 13 2"/></svg>'
                + '</div>'
                + '<span class="qs-label">' + escapeHtml(formatSkillName(s.name)) + '</span>'
                + (metaLine ? '<div style="margin-top: 0.25rem;">' + metaLine + '</div>' : '')
                + '<button class="qs-star" data-skill="' + escapeHtml(s.name) + '" onclick="toggleFavorite(\'' + escapeHtml(s.name).replace(/'/g, "\\'") + '\', event)" aria-label="Add to favorites">&#9733;</button>';
            grid.appendChild(card);
        });

        document.getElementById('recommended-section').style.display = 'block';
        updateStarStates();
    } catch (e) {
        // Non-critical, silently fail
        console.warn('Could not load recommendations:', e);
    }
}
